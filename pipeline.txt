pipeline{
    agent any
    stages{
        stage('Checkout'){
            steps{
                git url: 'https://github.com/Chiranthan04/ex-mvn.git', branch: 'main'
            }
        }
        stage('Build'){
            steps{
                sh '/usr/bin/mvn clean install'
            }
        }
        stage('Test'){
            steps{
                sh 'mvn test'
            }
        }
    }
    post{
        always{
            junit 'target/surefire-reports/*.xml'
        }
        success{
            echo 'Success'
        }
        failure{
            echo 'Failed'
        }
    }
}
